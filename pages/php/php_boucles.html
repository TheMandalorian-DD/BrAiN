Les Boucles

<?php
  	
	/* La boucle -> TantQue la condition est "Vraie"
	   On écécute les instructions
	FinTantQue */
	
	// Variable compteur : $i, $j, $k, ...
	
	// Exemple avec la boucle while (Tant que) compter de 0 à 10
	
	$nb = 0;
	
	while($i < 11){
	   echo $i;
	   $i++;
	}
	
	// Même exemple avec la boucle for
	
	for($i = 0; $i < 11; $i++){
	   echo $i;
	}
			
	// Exemple avec la do while, faire une action tant que la condition n'est pas valider

	do{
	   $jouer = readline("Jouer : oui ou non ?");
	}while($jouer != "non");
	
	/* break; peut être utilser pour stoper l'éxécution d'une boucle
	quand un certains résultat est atteint. */
	
?>
