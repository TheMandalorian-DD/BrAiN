Les Variables et Constantes

Variables

<?php
    
	// Les types
	
	Booléen (boolean, bool) : true (1) || false (0)
	Entier (integer, int) : 14 || -126, 32000
	Flottant (float, double) : 3.14 || -252.5
	Chaîne de caractères (string) : 'Hello' || "World"
	
	// Déclarons une variable
		
	$firstVariable = "Salut";
	$secondVariable = 50.5;	
			
	// La concaténation se fait avec un point !

	$var = "World";
	echo "Hello".$var."!";
		
?>

Constantes

<?php
    
	/* Une constante est une variable dont on ne peut
	     modifier la valeur une fois définit.
	Elle n'est accessible qu'en lecture ! */ 
	
	// Par convention une constante est définit en majuscule
	
	// Exemple :
	
	define("PI", 3.14);

	// Pour l'afficher :
		
	echo PI;
	
	// Autre exemple :
	
	$name = 'Dalil';
	
	define("USER_NAME", $name);
		
	echo USER_NAME;
		
?>
