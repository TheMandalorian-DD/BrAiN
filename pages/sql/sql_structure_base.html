Création d'une Database

CREATE DATABASE IF NOT EXISTS nomDB;
--- ou ---
CREATE SCHEMA nomDB;

Ex : CREATE DATABASE IF NOT EXISTS base1;
USE base1; #Use pour utiliser la base de données.

Création d'une Table

CREATE TABLE IF NOT EXISTS nomtable(
propriété1 type,
propriété2 type
);

Ex : CREATE TABLE IF NOT EXISTS nomtable(
id INT PRIMARY KEY AUTO_INCREMENT,
pseudo VARCHAR(255),
password TEXT
);

Suppression d'une Database

DROP DATABASE nomDB;
--- ou ---
DROP SCHEMA nomDB;

Suppression d'une Table

USE nomDB; #Sélectionner la base de données.
DROP TABLE nomTable;
Suppression de toutes les données d'une Table

USE nomDB; #Sélectionner la base de données.
TRUNCATE TABLE nomTable;

Modifier le nom d'une base de données

USE nomDB; #Sélectionner la base de données.
ALTER DATABASE nomDB RENAME nouveaunom;
--- ou ---
ALTER SCHEMA nomDB RENAME nouveaunom;

Modifier une Table

Ajouter une colonne :
USE nomDB; #Sélectionner la base de données.
ALTER TABLE nomtable ADD COLUMN nomcolonne Type;
Pour décider de l'emplacement de la colonne: "FIRST, BEFORE, AFTER..."
Ex : ALTER TABLE users ADD COLUMN id INT PRIMARY KEY AUTO_INCREMENT FIRST;
--- ou ---
ALTER TABLE users
ADD COLUMN pseudo VARCHAR(25) BEFORE date,
ADD COLUMN password TEXT AFTER pseudo;

Changer la position de la colonne :
Ex : ALTER TABLE users CHANGE COLUMN pseudo pseudo VARCHAR(25) AFTER date; Ajouter une clé étrangère :
Ex : ALTER TABLE users ADD FOREIGN KEY (users_id) REFERENCES user(id);

Suppression de la colonne clé étrangère :
Ex : ALTER TABLE users DROP INDEX users_id;
ALTER TABLE users DROP FOREIGN KEY;

Suppression d'une colonne :
ALTER TABLE nomTable DROP COLUMN nomcolonne;

Changer le nom d'une colonne :
ALTER TABLE nomTable CHANGE nomActuelle nouveauNom type;
